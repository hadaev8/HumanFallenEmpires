hfe_map_mode_plague = {
	icon = GFX_hfe_map_mode_plague
	enable_terra_incognita = yes
	can_change_point_of_view = no
	color = {
		value = {
			0
			0
			0.8
			0.7
		}
		condition = {
			NOT = {
				exists = FROM
			}
		}
		custom_tooltip = "MAP_MODE_OBSERVER"
	}
	color = {
		value = {
			0
			0
			0.8
			0.7
		}
		custom_tooltip = "MAP_MODE_NO_CONTACT"
		condition = {
			exists = from
			not = {
				has_established_contact = from
			}
		}
	}
	color = {
		value = {
			0
			0
			0.8
			0.7
		}
		condition = {
			hidden_trigger = {
				num_pops = 0
			}
		}
	}
	# have no plague
	color = {
		value = {
			0.25
			1
			0.6
			1
		}
		custom_tooltip = hfe_map_mode_plague_not_infected
		condition = {
			not = {
				any_owned_planet = {
					has_planet_flag = hfe_plague_planet
				}
			}
		}
	}
	# have plague
	color = {
		value = {
			0
			0.9
			0.6
			1
		}
		custom_tooltip = hfe_map_mode_plague_infected
		condition = {
			any_owned_planet = {
				has_planet_flag = hfe_plague_planet
			}
		}
	}
}
