hfe_wg_coalition_challenge_leadership = {
	casus_belli = hfe_cb_coalition_challenge_leadership
	hide_if_no_cb = yes
	surrender_acceptance = -100
	war_exhaustion = 1
	potential = {
		always = yes
	}
	possible = {
		always = yes
	}
	on_accept = {
		set_subject_of = {
			who = none
		}
		every_relation = {
			limit = {
				or = {
					is_subject_type = hfe_vassal_future
					is_country = from
				}
			}
			set_subject_of = {
				who = root
				subject_type = hfe_vassal_future
			}
		}
		save_global_event_target_as = hfe_coalition_leader
	}
	ai_weight = {
		weight = 1000
		modifier = {
			factor = 0
			relative_power = {
				who = from
				category = fleet
				value < equivalent
			}
		}
	}
	allowed_peace_offers = {
		surrender
		status_quo
		demand_surrender
	}
}
