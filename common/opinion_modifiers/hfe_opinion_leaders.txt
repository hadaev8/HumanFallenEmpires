				has_trait = hfe_ruler_trait_warriror_honor
			}
			is_homicidal = no
		}
	}
	opinion = {
		base = 20
	}
}

hfe_opinion_leaders_manipulative = {
	trigger = {
		is_country_type = default
		is_homicidal = no
		exists = from
		from = {
			exists = ruler
			ruler = {
				has_trait = hfe_ruler_trait_manipulative
			}
			not = {
				is_rival = prev
			}
			is_homicidal = no
		}
	}
	opinion = {
		base = 10
	}
}

hfe_opinion_leaders_manipulative_rival = {
	trigger = {
		is_country_type = default
		is_homicidal = no
		exists = from
		from = {
			any_relation = {
				exists = ruler
				ruler = {
					has_trait = hfe_ruler_trait_manipulative
				}
				is_rival = prev
				is_homicidal = no
			}
		}
	}
	opinion = {
		base = -10
	}
}
