hfe_army_high_mechanicus_bodyguards = {
	defensive = yes
	is_pop_spawned = no
	pop_limited = no
	health = 1.1
	damage = 2
	morale = 5
	has_morale = yes
	morale_damage = 1.5
	collateral_damage = 0.9
	war_exhaustion = 2
	icon_frame = 5
	has_species = yes

	resources = {
		category = armies
		cost = {
#			trigger = {
#				owner = {
#					NOT = {
#						is_primitive = yes
#						is_country_type = primitive
#					}
#				}
#			}
			energy = 500
			food = 30
			alloys = 300
		}
#		cost = {
#			alloys = 100
#		}
		produces = {
			unity = 1
		}
		upkeep = {
			food = 2
			alloys = 2
			energy = 2
			minerals = 2
		}
	}

	potential = {
		owner = {
			has_country_flag = hfe_varelse_primitive_0 

			OR = {
				has_civic = hfe_civic_high_mechanicus
				has_country_flag = high_mechanicus_retained
				is_spiritualist = yes
			}
			NOT = {
				has_civic = hfe_civic_overthrown_mechanicus
				has_country_flag = hfe_varelse_fedora_governance
				is_materialist = yes
			}
		}
		planet = {
			OR = {
				has_planet_flag = hfe_colony_beta
				is_capital = yes
			}
		}
	}
	allow = {
		planet = {
			NOT = { # Only one can exist on a planet
				any_planet_army = {
					army_type = hfe_army_high_mechanicus_bodyguards
				}
			}
		}
		OR = {
			FROM = {
				OR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
					is_robotic = yes
				}
			}
			owner = { has_country_flag = hfe_varelse_primitive_0 }
		}
		OR = {
			FROM = {
				OR = {
					has_citizenship_rights = yes
					has_citizenship_type = { type = citizenship_full }
					is_same_species = owner
					is_same_species = event_target:hfe_varelse_species
				}
			}
			owner = { is_primitive = yes }
			owner = { is_country_type = primitive }
		}
		owner = { # Can only be re-spawned in peacetime
			is_at_war = no
		}
	}
}
