# pirate life simulation
country_event = {
	id = human_fallen_empires_pirate.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = human_fallen_empires_pirate
		exists = capital_scope
	}

	immediate = {
		every_owned_planet = {
			limit = {
				is_asteroid = yes
			}
			log = "fixing planet class for [this.GetName], owner: [owner.GetName]"
			change_pc = human_fallen_empires_pc_pirate_base
		}
		# move capital if have normal planet
		if = {
			limit = {
				capital_scope = {
					is_planet_class = human_fallen_empires_pc_pirate_base
				}
				any_owned_planet = {
					not = {
						is_planet_class = human_fallen_empires_pc_pirate_base
					}
				}
			}
			random_owned_planet = {
				limit = {
					not = {
						is_planet_class = human_fallen_empires_pc_pirate_base
					}
				}
				set_capital = yes
			}
		}
		# build new bases
		country_event = {
			id = human_fallen_empires_pirate.3
		}
		# send fleet logic for unstuck fleets
		every_owned_fleet = {
			limit = {
				num_ships > 1
				is_ship_class = shipclass_military
			}
			fleet_event = {
				id = human_fallen_empires_pirate.300
			}
		}
		# try to hunt weaker near pirate
		random_relation = {
			limit = {
				is_country_type = human_fallen_empires_pirate
				relative_power = {
					who = root
					category = fleet
					value < equivalent
				}
				or = {
					is_neighbor_of = root
					any_system_within_border = {
						distance_to_empire = {
							who = root
							distance <= 20
						}
					}
				}
				not = {
					has_opinion_modifier = {
						who = root
						modifier = human_fallen_empires_opinion_pirate_prior_target
					}
				}
			}
			add_opinion_modifier = {
				who = root
				modifier = human_fallen_empires_opinion_pirate_prior_target
			}
			#log = "pirate try to hunt pirate hunter: [prev.GetName] target: [this.GetName]"
		}
		country_event = {
			id = human_fallen_empires_pirate.10
			days = 1
			random = 360
		}
		country_event = {
			id = human_fallen_empires_pirate.11
			days = 1
			random = 360
		}
	}
}

					add_energy = @human_fallen_empires_pirate_base_cost_minus
