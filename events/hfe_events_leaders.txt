	id = hfe_leaders.0
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		hfe_trigger_leaders_recrutment_enabled = yes

			is_ai = no
			is_country_type = default
		}
		any_pool_leader = {
			always = yes
		}

	immediate = {
		if = {
			limit = {
				not = {
					has_modifier = hfe_leaders_started
				}
			}
			add_modifier = {
				modifier = hfe_leaders_started
			}
		}
		hfe_leaders_calc_leaders_capacity = yes
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = governor
					}
				}
			}
			country_event = {
				id = hfe_leaders.50
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = scientist

					}
				}
			}
			country_event = {
				id = hfe_leaders.51
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = admiral
					}
				}
			}
			country_event = {
				id = hfe_leaders.52
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = general
					}
				}
			}
			country_event = {
				id = hfe_leaders.53
			}
		}
	}
}

# gain new trait from mod by on_leader_level_up
country_event = {
	id = hfe_leaders.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		exists = from
		hfe_trigger_leaders_traits_enabled = yes
	}
	immediate = {
		if = {
			limit = {
				is_country_type = hfe_pirate
				from = {
					leader_class = admiral
					not = {
						has_trait = hfe_admiral_trait_pirate
					}
				}
			}
			from = {
				add_trait = hfe_admiral_trait_pirate
			}
		}
		from = {
			if = {
				limit = {
					species = {
						has_trait = trait_psionic
					}
				}
				hfe_effect_add_psionic_trait = yes
			}
			add_leader_traits_after_modification = yes
			if = {
				limit = {
					num_traits < 4
				}
				random_list = {
					90 = {
					}
					10 = {
						modifier = {
							factor = 2
							has_level > 3
						}
						modifier = {
							factor = 2
							has_level > 4
						}
						modifier = {
							factor = 2
							has_level > 5
						}
						hfe_effect_leaders_add_random_trait = yes
					}
				}
			}
		}
	}
}

	is_triggered_only = yes

		check_variable = {
			which = hfe_leaders_governor_capacity
			value > 0
		}
	}

		random_pool_leader = {
			limit = {
				leader_class = governor

				clone_leader = {
					target = prev

				}

			kill_leader = {
				show_notification = no
			}

			which = hfe_leaders_governor_capacity
			value = -1
		}
		country_event = {
			id = hfe_leaders.54
		}
	}
}

# fixing leaders traits
country_event = {
	id = hfe_leaders.104
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		exists = from
		exists = from.species
	}
	immediate = {
		from = {
			if = {
				limit = {
					leader_class = admiral
					species = {
						has_trait = hfe_species_trait_dolphin
					}
				}
				add_trait = hfe_admiral_trait_dolphin
			}
			if = {
				limit = {
					leader_class = admiral
					root = {
						or = {
							has_country_flag = hfe_future
							and = {
								exists = event_target:hfe_coalition_leader
								is_country = event_target:hfe_coalition_leader
							}
						}
					}
				}
				add_trait = hfe_admiral_trait_future_train
			}
		}
		# civic bonus
		if = {
			limit = {
				has_valid_civic = hfe_civic_features_designed_aristocracy
			}
			from = {
				hfe_effect_features_designed_aristocracy = yes
			}
		}
	}
}
