# Total War and The War in Heavens by Human Fallen Empires
namespace = hfe_total_war
# star war events 0 - 50
# invite subjects and set flags
# on_war_beginning
# A war is beginning, executed for every country in the war.
# Root = Country
# From = War
country_event = {
	id = hfe_total_war.0
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		from = {
			using_war_goal = {
				type = hfe_wg_total_war
				owner = root
			}
		}
	}
	immediate = {
		if = {
			limit = {
				is_war_leader = yes
			}
			if = {
				limit = {
					has_global_flag = hfe_general_logs
				}
				log = "hfe total war run core systen flags for [this.GetName]"
			}
			every_system_within_border = {
				set_star_flag = hfe_total_war_core_system_of@root
			}
		}
		if = {
			limit = {
				is_overlord = yes
			}
			if = {
				limit = {
					has_global_flag = hfe_general_logs
				}
				log = "hfe total war call subjects for [this.GetName]"
			}
			every_subject = {
				join_war = root
			}
		}
	}
}

# annex occupying systems planet 51
# A planets controller becomes a country not the same as the owner.
# Root = Planet
# From = Planet Owner
# FromFrom = Planet Controller (the one occupying)
planet_event = {
	id = hfe_total_war.51
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		hfe_trigger_war_in_heavens_custom_wg_system_shift_enabled = yes
		exists = owner
		exists = controller
		exists = solar_system
		nor = {
			is_controlled_by = owner
			solar_system = {
				or = {
					has_star_flag = hfe_total_war_core_system_of@owner
					any_system_planet = {
						# colonized planet
						is_owned_by = prev.owner
						is_controlled_by = prev.owner
					}
				}
			}
		}
		# this is total war
		owner = {
			any_war = {
				# any_war_participant = {
				# 	is_country = root.owner
				# 	is_war_leader = no
				# }
				any_war_participant = {
					is_country = root.controller
				}
				# total war used
				any_war_participant = {
					prev = {
						using_war_goal = {
							type = hfe_wg_total_war
							owner = prev
						}
					}
					# or = {
					# 	is_in_federation_with = root.controller
					# 	is_country = root.controller
					# 	any_subject = {
					# 		is_country = root.controller
					# 	}
					# }
				}
			}
		}
	}
	immediate = {
		if = {
			limit = {
				has_global_flag = hfe_general_logs
			}
			log = "hfe total war change owner of planets and system [solar_system.GetName] old owner [owner.GetName] new owner [controller.GetName]"
		}
		solar_system = {
			every_system_planet = {
				limit = {
					is_owned_by = root.owner
				}
				# conquer = controller
				set_owner = controller
			}
		}
		if = {
			limit = {
				exists = starbase
			}
			starbase = {
				set_owner = root.controller
			}
		}
	}
}
