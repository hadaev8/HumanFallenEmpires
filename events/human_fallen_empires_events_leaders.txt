country_event = {
	id = human_fallen_empires_leaders.0
	hide_window = yes
	is_triggered_only = yes

		human_fallen_empires_trigger_leaders_recrutment_enabled = yes

			is_ai = no

			always = yes
		}

		if = {
			limit = {
				not = {
					has_modifier = human_fallen_empires_leaders_started

				modifier = human_fallen_empires_leaders_started
				days = -1
			}
		}
		human_fallen_empires_leaders_calc_leaders_capacity = yes

			limit = {
				not = {
					any_owned_leader = {
						leader_class = governor
					}
				}
			}
			country_event = {
				id = human_fallen_empires_leaders.50
			}
		}
		if = {
			limit = {
				not = {
					any_owned_leader = {
						leader_class = scientist
					}
				}
			}
			country_event = {
				id = human_fallen_empires_leaders.51
			}
		}
		if = {
			limit = {
				not = {

# gain new trait from mod by on_leader_level_up
country_event = {
	id = human_fallen_empires_leaders.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		human_fallen_empires_trigger_leaders_traits_enabled = yes
	}

	immediate = {
		from = {
			if = {
				limit = {
					species = {
						has_trait = trait_psionic
					}
				}
				human_fallen_empires_effect_add_psionic_trait = yes
			}
			add_leader_traits_after_modification = yes
			if = {
				limit = {
					num_traits < 4
				}
				random_list = {
					90 = {
					}
					10 = {
						modifier = {
							factor = 2
							has_level > 3
						}
						modifier = {
							factor = 2
							has_level > 4
						}
						modifier = {
							factor = 2
							has_level > 5
						}
						human_fallen_empires_effect_leaders_add_random_trait = yes
					}
				}
			}
		}
	}
}

	hide_window = yes
	is_triggered_only = yes

	trigger = {
		check_variable = {
			which = human_fallen_empires_leaders_governor_capacity
			value > 0
		}
	}

	immediate = {
		random_pool_leader = {
			limit = {
				leader_class = governor
			}
			root = {
				clone_leader = {
					target = prev
				}
			}
			kill_leader = {
				show_notification = no
			}
		}
		change_variable = {
			which = human_fallen_empires_leaders_governor_capacity
			value = -1
		}
		country_event = {
			id = human_fallen_empires_leaders.54
		}
	}
}
