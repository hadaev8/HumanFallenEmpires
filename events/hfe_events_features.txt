# biobomb feature
planet_event = {
	id = hfe_features.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		habitable_planet = yes
		exists = owner
		owner = {
			has_ascension_perk = hfe_ap_features_biobomb
		}
	}
	immediate = {
		create_species = {
			class = random
			blocked_archetypes = {
				MACHINE       
				ROBOT       
				LITHOID
			}
			homeworld = root
		}
		owner = {
			if = {
				limit = {
					has_trait = trait_hive_mind
				}
				modify_species = {
					species = last_created_species
					add_trait = trait_hive_mind
				}
				last_created_species = {
					set_citizenship_type = {
						country = prev
						type = citizenship_full
						cooldown = no
					}
				}
			}
		}
		owner = {
			if = {
				limit = {
					has_country_flag = flesh_weakened
					NOT = { has_country_flag = synthetic_empire }
				}
				modify_species = {
					species = last_created_species
					add_trait = trait_cybernetic
				}
			}
		}
		owner = {
			if = {
				limit = {
					OR = {
						has_trait = trait_latent_psionic
						has_ascension_perk = ap_mind_over_matter
					}
					NOT = {	has_ascension_perk = ap_transcendence }
				}
				modify_species = {
					species = last_created_species
					add_trait = trait_latent_psionic
				}
			}
		}
		owner = {
			if = {
				limit = {
					OR = {
						has_trait = trait_psionic
						has_ascension_perk = ap_transcendence
					}
				}
				modify_species = {
					species = last_created_species
					add_trait = trait_psionic
				}
			}
		}
		random_list = {
			60 = {
			}
			40 = {
				hfe_effect_add_random_trait = {
					hfe_species = last_created_species
				}
			}
		}
		create_pop = {
			species = last_created_species
		}
		last_created_species = {
#			set_species_flag = syncretic_species
			set_species_flag = hfe_artificial_species
		}
	}
}
